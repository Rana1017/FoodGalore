-----------TRIGGER
CREATE OR REPLACE TRIGGER USERS
BEFORE INSERT ON USERS
FOR EACH ROW
BEGIN
    IF :new.USER_ID IS NULL THEN
        SELECT users_seq.nextval INTO :new.USER_ID FROM DUAL;
    END IF;
END;
